108 /** @brief Handles the PWM1 parameter interrupts. \n
109  * We use two sources of PWM1 interrupts:
110  * - on the right flank of slice2 (after settling of motor current): read Imot.
111  * - on the right flank of slice1 (H-bridge turned off): read Back EMF.
112  */
113 void __interrupt (irq(IRQ_PWM1), base(IVT1_BASE_ADDRESS), high_priority)
114 PWM1_isr (void)
115 {
116  int16_t mAmps;
117  uint16_t uk;
118 
119  // cause of interrupt: slice 1 parameter 2?
120  if (PWM1GIRbits.S1P2IF) // 2 ms after H-bridge ON
121  {
122  PWM1GIRbits.S1P2IF = 0; // clear interrupt flag
123 
124  ina219_reg(1); // exec time ca. 49 ¬µs (@SCL 400 kHz)
125  mAmps = ina219_read(); // exec time ca. 72 ¬µs (@SCL 400 kHz)
126  if (mAmps < 0) mAmps = 0; // offset may cause negative readings ??
127 
128 #ifdef USE_mAMP_LoPASS
129  g_mAx10 += (mAmps - g_mAx10) >> NLoPASS_CURR; // is usually > 0
130 #else
131  g_mAx10 = mAmps;
132 #endif
133  if(g_ns_curr < LOGSIZE)
134  {
135  g_curr_log[g_ns_curr++] = (uint8_t)(g_mAx10 >> 2);
136  }
137 #ifdef TEST_mAMPS2DAC
138  DAC1DATL = (uint8_t) (g_mAx10 >> 2); // current monitor (16 -> 8 bit)
139 #endif
140  } // slice 1 parameter 2
141 
142 
143 
144 
145  // cause of interrupt: slice 1 parameter 1?
146  if (PWM1GIRbits.S1P1IF) // immediately after H-bridge OFF
147  {
148  PWM1GIRbits.S1P1IF = 0; // clear interrupt flag
149 
150  __delay_us(400); // allow vbemf to stabilize
151  /** @todo This 400 ¬µs delay can cause overrun in the UART!
152  * At 4800 Bd one char is received every 260 ¬µs.
153  * Generate a synchronized PWM to get a perfect timing!
154  */
155 
156  uk = daq_vbemf(g_vz); // Exec time ca. 100 ¬µs
157 
158 #ifdef USE_BEMF_LoPASS
159  // see: PIC datasheet 40.5.6 Low-Pass Filter Mode and microchip AN2749
160  g_vbemf += (uk - g_vbemf) >> NLoPASS_BEMF; // is usually > 0
161 #else
162  g_vbemf = uk;
163 #endif
164  // data logger: save g_vbemf into bemf_log[]
165  if(g_ns_bemf < LOGSIZE)
166  {
167  g_vbemf_log[g_ns_bemf++] = (uint8_t)(g_vbemf >> 4);
168  }
169 
170 #ifdef TEST_VBEMF2DAC
171  // monitor VBEMF (16 -> 8 bit) - requires DACout routed to a pin
172  DAC1DATL = (uint8_t) (g_vbemf >> 4);
173 #endif
174  // sum up speed
175  if (g_dir > 0) g_vbemf_sum[g_vz] += g_vbemf;
176  else if (g_dir < 0) g_vbemf_sum[g_vz] -= g_vbemf;
177 
178  } // slice 1 parameter 1
179 
180 } // PWM1_isr ()

228 /** @brief Handles UART receive events.
229  * - Interrupt when receiving data from ESP via the UART
230  * - Stores chars in g_rx232_buf[]
231  * - Sets flag g_rs232_request = 1 after detection of CR or LF.
232  */
233 void __interrupt (irq(IRQ_U1RX), base(IVT1_BASE_ADDRESS), low_priority)
234 U1RX_isr (void)
235 {
236  uint8_t ch; // received ascii char
237 
238  // PIR4bits.U1RXIF = 0; // U1RXIF cannot be cleared by software
239 
240  if (U1ERRIRbits.RXFOIF || U1ERRIRbits.FERIF)
241  { // overrun or framing error?
242  U1ERRIRbits.RXFOIF = U1ERRIRbits.FERIF = 0;
243  ch = U1RXB;
244  init_uart1(); // reset serial port
245  }
246  else
247  { // valid char received, save to buffer, check if CR
248  ch = U1RXB;
249  if (ch == '\r' || ch == '\n')
250  {
251  ch = 0;
252  if (g_rx232_count > 1)
253  {
254  g_rs232_request = 1;
255  PIE4bits.U1RXIE = 0; // disable further U1RX interrupts
256  } // until cmd has been processed
257  else // ignore leading CRLF
258  {
259  goto _exit;
260  }
261  } // EOL received
262  if (g_rx232_count < sizeof(g_rx232_buf))
263  {
264  g_rx232_buf[g_rx232_count++] = ch; // save to buffer
265  }
266  else // buffer overflow, flush RS232 buf
267  {
268  for (uint8_t i = 0; i < sizeof(g_rx232_buf); i++) g_rx232_buf[i] = 0;
269  g_rx232_count = 0;
270  }
271  } // valid char
272 _exit:
273  NOP();
274 
275 } // U1RX_isr()
