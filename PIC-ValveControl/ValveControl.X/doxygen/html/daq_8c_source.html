18 /** @brief Returns an uncalibrated temperature estimation. \n
19  * Uses the temperature indicator (TI) of the PIC µC.
20  * Before calling this function, FVR and TI must have been
21  * initialized and enabled. \n
22  * Execution time ca. 890 µs (incl. 8-fold averaging).
23  * @return int16_t temperature [0.1 °C] (-2731 on ADC error)
24  */
25 int16_t
26 daq_temperature (void)
27 {
28  int8_t error = 0;
29  int24_t tempC = -2731;
30  int32_t sum;
31  int16_t TSHR1; // Temp. indicator gain (hi range)
32  int16_t TSHR3; // Temp. indicator offset (hi range)
33 
34  NVMADR = 0x2C002A; // @TSHR1 = Gain, high range
35  NVMCON1bits.CMD = 0b000; // read command (PFM: word)
36  NVMCON0bits.GO = 1; // start word read
37  while (NVMCON0bits.GO); // wait for the read operation to complete
38  TSHR1 = (int16_t) NVMDAT; // 0xFDEE = -530
39 
40  NVMADR = 0x2C002E; // @TSHR3 = Offset (high range)
41  NVMCON1bits.CMD = 0x00; // set the word read command
42  NVMCON0bits.GO = 1; // start word read
43  while (NVMCON0bits.GO); // wait for the read operation to complete
44  TSHR3 = (int16_t) NVMDAT; // 0x1A03 = 6659
45 
46  adc_init (0x3C); // ADPCH = Temp. Indicator, VREF+ = ADFVR (2x)
47  sum = 0;
48  for (uint8_t i = 0; (i < 8) && (error == 0); i++)
49  { // averaging (8x)
50  adc_start(0x3C);
51  error = adc_wait();
52  sum += ADRES;
53  }
54 
55  if (error == 0)
56  { // temp = ((ADRES*Gain)/256 + Offset) / 10; get Offset & Gain from DIA
57  sum >>= 3;
58  tempC = (int24_t) (sum) * TSHR1; // ADRES * gain
59  tempC >>= 8;
60  tempC += TSHR3;
61  //tempC /= 10; don't divide by 10 - we return in 1/10th degree
62  }
63  return ((int16_t) tempC);
64 
65 } // daq_temperature()

68 /** @brief Measures the voltage VDD of the controller \n
69  * With ADC_VREF+ = VDD and ADC_Positive_Input = FVR1 we get: \n
70  * u(VFR1) = ADRES * Vdd / 4095 := FVRA2X / 1000 => \n
71  Vdd = FVR2AX * 4095 / 1000 / ADRES ; in [V]
72  * @return Voltage [0.01 V]
73  */
74 uint16_t
75 daq_vdd (void)
76 {
77  uint32_t v;
78  uint16_t vdd;
79 
80  adc_init(0b111110); // ADC Positive Input = FVR1 (ADC module)
81  ADREF = 0; // change VREF+ = VDD
82  adc_start(0b111110);
83 
84 #ifdef VDD_WITHOUT_DIA
85  // Measured value Eval Board using default FVR = 2.048 V: 3,29 V
86  // Vref = adc * Vdd / 4095 = 2.048 V =>
87  // Vdd = 4095 * 2.048 / adc = 8386,56 / adc
88  if (adc_wait() < 0) vdd = 999; // signal error: 9.99 V
89  else vdd = (uint16_t) (838656L / ADRES); // [0.01 V]
90 #else
91  // Measured value Eval Board using FVRA2X from DIA: 3,32 V
92  if (adc_wait() < 0) vdd = 999; // signal error: 9.99 V
93  else
94  {
95  v = (uint32_t) FVRA2X * 4095 / 10;
96  vdd = (uint16_t) (v / ADRES); // in [0.01 V]
97  }
98 #endif
99  return (vdd);
100 
101 } // daq_vdd ()

105 /** @brief Reads the back EMF voltage of channel vz. \n
106  * Returns the raw ADC RESULT. \n
107  * Exec time ca. 100 µs
108  * @return Voltage [ADC raw]

110 uint16_t
111 daq_vbemf (uint8_t vz)

114  uint32_t uval32;

117  switch (vz)

137  else vbemf = ADRES; // Basic mode

139  // read twice and average

141  if (adc_wait() < 0) vbemf += g_vbemf; // if timeout: keep previous value
142  else vbemf += ADRES;

144
