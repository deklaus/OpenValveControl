285 /** @brief Init the PWM (125 Hz).
286  * - 90% duty cycle (7.2 ms + 0.8 ms = 8 ms) \n
287  see datasheet 28.4.1f.
288 */
289 void init_pwm (void)
290 {
291 /* Use Timer2 and configure the default period.
292  * PR2 = FREQUENCY_TO_PR_CONVERT(FREQUENCY_MIN);
293  * PWM Period = (T2PR + 1) x Tosc x TMR2_Prescaler ; Tosc = 4/Fosc = 250 ns
294  * = (T2PR + 1) x 32 ¬µs ; CKPS = 128
295  * max.Period = 256 x 32 ¬µs = 8,192 ms = 122 Hz
296  * min.Period = 1 x 32 ¬µs = 32 ¬µs = 31,25 kHz
297  */
298  T2CONbits.ON = 0; // TIMERx stop during setup
299  T2TMR = 0x00; // TIMERx counter reset (forced by stop)
300  T2CLKCONbits.CS = 1; // TIMERx clock source is FOSC/4
301 
302  // Configure the PWM period
303  T2PR = FREQUENCY_TO_PR_CONVERT(125L); // PWM -> 125 Hz (T2PR = 249)
304  T2CONbits.CKPS = 0b111; // TIMERx prescaler 1:128
305  T2CONbits.OUTPS = 0; // TIMERx postscaler 1:1
306  T2CONbits.ON = 1; // TIMERx ON
307 
308  /* - Configure the CCP module for the PWM mode by loading the CCPxCON
309  * register with the appropriate values.
310  * - Load the CCPRx register with the PWM duty cycle value and configure
311  * the FMT bit to set the proper register alignment.
312  */
313  CCP1CONbits.MODE = 0x0C; // MODE PWM
314  CCP1CONbits.FMT = 1; // FMT left_aligned
315  CCP1CONbits.EN = 1; // EN enabled
316  CCPTMRS0bits.C1TSEL = 1; // Selecting Timer 2
317 
318  // Configure the PWM duty cycle (percent))
319  CCPR1 = (DUTYCYCLE_TO_CCPR_CONVERT(90, 125)) << 6;
320 
321 } // init_pwm ()
