49 /** @brief Read 2 byte of data from INA219. \n
50  * NOTE: register must be set prior read! \n
51  * When ABD is set (ABD = 1), the address buffer is disabled. In this case,
52  * the number of data bytes are loaded into I2CxCNT, and the client’s 7-bit
53  * address and R/W bit are loaded into I2CxTXB. A write to I2CxTXB will cause
54  * host hardware to automatically issue a Start condition once the bus is
55  * Idle (BFRE = 1). Software writes to the Start bit are ignored. \n
56  * See datasheet (36.4.2.7.1), page 635.
57  */
58 int16_t ina219_read (void)
59 {
60  int16_t value = 0;
61 
62  for (uint8_t timeout = 0; !I2C1STAT0bits.BFRE && (++timeout < 50); );
63  if (I2C1STAT0bits.BFRE)
64  {
65  I2C1STAT1bits.TXWE = 0; // Clear Transmit Write Error Status
66  I2C1PIRbits.PCIF = 0;
67  if (I2C1STAT1bits.RXBF) value = I2C1RXB; // Clear rxbuf
68  I2C1STAT1 = 0; // clear error status
69 
70  I2C1CNTL = 0x02; // Count register
71  I2C1CNTH = 0x00;
72 
73  // Transmit the INA's I2C address (0x40 << 1) + READ
74  I2C1TXB = I2C_ADDR_INA219 + I2C_READ; // address + read
75 
76  // Receive the INA's Register content (MSB first)
77  for (uint8_t timeout = 0; !I2C1STAT1bits.RXBF && (++timeout < 100); );
78  value = ((int16_t) I2C1RXB) << 8;
79 
80  for (uint8_t timeout = 0; !I2C1STAT1bits.RXBF && (++timeout < 100); );
81  value += I2C1RXB; // LB
82 
83  // wait 'til transfer has completed (stop condition detected)
84  for (uint8_t timeout = 0; !I2C1PIRbits.PCIF && (++timeout < 50); );
85  // a minimum delay is required following every write_i2c() !
86  __delay_us(10); // min 4 µs
87 
88  }
89  else
90  { // error handler
91  value = 0xFFBB; // "bus busy", handle negative values by caller!
92  }
93 
94  return(value);
95 
96 } // ina219_read()
