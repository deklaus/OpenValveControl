454 /** @brief Check for over current (e.g. due to blocked drive)
455  * - Compares actual current g_mAx10 with limit of selected valve zone vz
456  * - On overcurrent, counter n_overcurr gets incremented, else counter is reset
457  * - If counter exceeds 12 (ca. 100 ms),
458  * - PWM is stopped, g_dir forced to 0
459  * - Error flag OVER_CURR is set
460  * @param vz: valve zone [0, 1 - 4]
461  * @return 0: no overcurrent, 1: overcurrent detected
462  */
463 static bool over_current (uint8_t vz)
464 {
465  bool result = false;
466 
467  if (g_mAx10 > g_max_mAx10[vz]) // if over current
468  {
469  if (++n_overcurr > 12) // lasts longer than 100 ms
470  {
471  set_pwm(vz, 0); // stop pwm
472  g_dir = 0;
473  g_ERRORflags.OVER_CURR = 1;
474  result = true;
475  }
476  }
477  else n_overcurr = 0; // reset spike counter
478 
479  return (result);
480 
481 } // over_current ()
